// Hook for generating QR codes
export function useQRCodeGenerator() {
  const [isOpen, setIsOpen] = useState(false)
  const [qrData, setQrData] = useState<QRCodeData | null>(null)
 
  const generateQRCode = (data: QRCodeData) => {
    setQrData(data)
    setIsOpen(true)
  }
 
  const closeQRCode = () => {
    setIsOpen(false)
    setQrData(null)
  }
 
  return {
    isOpen,
    qrData,
    generateQRCode,
    closeQRCode
  }
}
